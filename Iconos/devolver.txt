  <script>
    const buttons = document.querySelectorAll('.tab-link');
    const contentBox = document.getElementById('tab-content');

    buttons.forEach(button => {
      button.addEventListener('click', () => {
        // Activar botón actual
        buttons.forEach(btn => btn.classList.remove('active'));
        button.classList.add('active');

        // Obtener archivo PHP del botón
        const tabFile = button.getAttribute('data-tab');

        // Cargar contenido con AJAX
        fetch(tabFile)
          .then(response => {
            if (!response.ok) throw new Error('Error al cargar contenido.');
            return response.text();
          })
          .then(html => {
            contentBox.innerHTML = html;
          })
          .catch(err => {
            contentBox.innerHTML = `<p style="color:red;">Error al cargar el contenido.</p>`;
            console.error(err);
          });
      });
    });
  </script>
